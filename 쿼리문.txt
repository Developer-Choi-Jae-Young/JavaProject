Create Table User (
	id INT(11) NOT NULL AUTO_INCREMENT,
	userid VARCHAR(20) NOT NULL,
	name VARCHAR(20) NOT NULL,
	password VARCHAR(80) NOT NULL,
	age INT(11) NOT NULL,
	gender VARCHAR(1) NOT NULL,
	phone VARCHAR(20) NOT NULL,
	email VARCHAR(20) NOT NULL,
	type VARCHAR(20) NOT NULL,
	CONSTRAINT Teacher_PK PRIMARY KEY(id)
);

Create Table RequestRegistUser (
	id INT(11) NOT NULL AUTO_INCREMENT,
	userid VARCHAR(20) NOT NULL,
	name VARCHAR(20) NOT NULL,
	password VARCHAR(80) NOT NULL,
	age INT(11) NOT NULL,
	gender VARCHAR(1) NOT NULL,
	phone VARCHAR(20) NOT NULL,
	email VARCHAR(20) NOT NULL,
	type VARCHAR(20) NOT NULL,
	CONSTRAINT RequestRegistUser PRIMARY KEY(id)
);

Create Table SelfStudy (
	id INT(11) NOT NULL AUTO_INCREMENT,
	student_id VARCHAR(20) NOT NULL,
	place VARCHAR(20) NOT NULL,
	teacher_id VARCHAR(20) NOT NULL,
	date DATETIME NOT NULL,
	CONSTRAINT SelfStudy_PK PRIMARY KEY(id),
	CONSTRAINT SelfStudy_FK2 FOREIGN KEY(student_id) REFERENCES User(id) ON DELETE CASCADE ON UPDATE CASCADE
);
